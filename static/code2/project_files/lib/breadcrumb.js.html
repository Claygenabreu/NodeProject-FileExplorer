//require node modules
const path = require('path');

const buildBreadcrumb = pathname => {
    //remove the first space in pathname:
    const cleanedPathname = pathname.substring(1);
    console.log(`cleanedPathname: ${cleanedPathname}`);

    //get all chunks inside pathname
    const pathChunks = (cleanedPathname).split('/');
    let breadcrumb = (`
  • Home
    • `);
    let link = '/';
    pathChunks.forEach((item, index) => {
        console.log(`link:${link}`);
        console.log(`item:${item}`);
        link = path.join(link, item);
        console.log(`joined link:${link}`);
        if(index !== pathChunks.length -1){
            breadcrumb += (`
  • ${item}
    • `);
        }else{
            breadcrumb += (`
  • ${item}
    • `);
        }
    });
    return breadcrumb;
};

module.exports = buildBreadcrumb;
